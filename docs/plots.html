prob <- function(C, P, tau, D) {
  profit <- (D - (C / (P / 1000))) / sqrt(tau) # loss
  prob_gain <- pnorm(profit)
  prob_gain
}



for (i in 1:length(mofo$C)) {
  mofo$breacking_even[i] <- prob(C = mofo$C[i], D = mofo$D[i], tau = mofo$tau[i], P = mofo$P[i]) * 100
}
